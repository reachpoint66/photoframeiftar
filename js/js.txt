document.addEventListener("DOMContentLoaded", function() {
    const frameSelector = document.getElementById("frameSelector");
    const photoFrame = document.getElementById("photoFrame");
    const imageInput = document.getElementById("imageInput");
    const downloadBtn = document.getElementById("downloadBtn");
    const canvas = document.getElementById("uploadedCanvas");
    const ctx = canvas.getContext("2d");

    // Tukar bingkai berdasarkan pilihan pengguna
    frameSelector.addEventListener("change", function() {
        let selectedFrame = this.value;
        photoFrame.src = "https://reachpoint66.github.io/photoframeiftar/assets/" + selectedFrame;
    });

    // Muat naik gambar pengguna ke atas bingkai
    imageInput.addEventListener("change", function(event) {
        const file = event.target.files[0];
        if (file) {
            const reader = new FileReader();
            reader.onload = function(e) {
                const img = new Image();
                img.src = e.target.result;
                img.onload = function() {
                    // Sesuaikan saiz canvas dan gambar
                    canvas.width = 1080;
                    canvas.height = 1080;
                    ctx.clearRect(0, 0, canvas.width, canvas.height);
                    ctx.drawImage(img, 0, 0, canvas.width, canvas.height);
                    ctx.drawImage(photoFrame, 0, 0, canvas.width, canvas.height);
                };
            };
            reader.readAsDataURL(file);
        }
    });

    // Muat turun gambar dengan bingkai
    downloadBtn.addEventListener("click", function() {
        const link = document.createElement("a");
        link.download = "photo_with_frame.png";
        link.href = canvas.toDataURL("image/png");
        link.click();
    });
});
